// batch file renamer — python 3.x
import os, re
from pathlib import Path

def batch_rename(folder: str, pattern: str, replace: str):
    p = Path(folder)
    for f in p.iterdir():
        if f.is_file():
            new_name = re.sub(pattern, replace, f.name)
            f.rename(p / new_name)
            print(f" {f.name} → {new_name}")

# Usage
batch_rename("./render_output", r"frame_(\d+)", r"shot_\1")
